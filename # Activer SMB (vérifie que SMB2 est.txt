# Activate SMB2
Set-SmbServerConfiguration -EnableSMB2Protocol $true -Force
Write-Host "SMB2 is activated."

# Enable WinRM
Enable-PSRemoting -Force
Write-Host "WinRM is activated."

# Start the WinRM service if it's stopped
Start-Service WinRM
Write-Host "WinRM service started."

# Check the status of SMB2 and WinRM
Get-SmbServerConfiguration | Select-Object EnableSMB2Protocol
Get-Service WinRM | Select-Object Status
